digraph finite_state_machine {
	rankdir=LR;
	size="8,5"
	node [shape = doublecircle]; Final;
	node [shape = circle];	
	Start -> Moving_Up [1];
	Start -> Moving_Down [2];
	Start -> Standing [3];
	Start -> Standing [4];
	Start -> Final [5];
	Moving_Up -> Standing [3];
	Moving_Down -> Standing [3];
	Standing -> Moving_Up [1];
	Standing -> Moving_Down [2];
	Standing -> Standing [3];
	Standing -> Standing [4];
	Standing -> Final [5];
}
