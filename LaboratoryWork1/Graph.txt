	rankdir=LR;
	size="8,5"
	node [shape = doublecircle]; Final;

	Start -> Moving_Up [label = 1];
	Start -> Moving_Down [label = 2];
	Start -> Standing [label = 3];
	Start -> Standing [label = 4];
